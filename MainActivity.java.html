18   public class MainActivity extends AppCompatActivity {
19       private ArrayList<String> items;
20       private ArrayAdapter<String> itemsAdapter;
21       private ListView listView;
22       private CheckBox checkBox;
23       private Button button;
24   
25       public MainActivity(ArrayAdapter<String> itemsAdapter) {
26           this.itemsAdapter = itemsAdapter;
27   
28       }
29   
30       @Override
31       protected void onCreate(Bundle savedInstanceState) {
32           super.onCreate(savedInstanceState);
33           setContentView(R.layout.activity_main);
34   
35           checkBox = findViewById(R.id.checkBox4);
36           button = findViewById(R.id.button1);
37   
38           button.setOnClickListener(new View.OnClickListener() {
39               @Override
40               public void onClick(View view) {
41                   addItem(view);
42               }
43           });
44   
45           items = new ArrayList<>();
46           itemsAdapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, items);

49

51       }
52   
53       private void setUpListViewListener() {
54           listView.setOnItemLongClickListener(new AdapterView.OnItemLongClickListener() {
55               @Override
56               public boolean onItemLongClick(AdapterView<?> parent, View view, int position, long id) {
57                   Context context = getApplicationContext();
58                   Toast.makeText(context, "Hapus", Toast.LENGTH_LONG).show();
59   
60                   items.remove(id);
61                   itemsAdapter.notifyDataSetChanged();
62                   return true;
63               }
64           });
65   
66       }
67   
68       private void addItem(View view) {
69           EditText input = findViewById(R.id.editTextTextPersonName4);
70           String itemText = input.getText().toString();
71   
72           if (!(itemText.equals(""))) {
73               itemsAdapter.add(itemText);
74               input.setText("");
75   
76           } else {
77               Toast.makeText(getApplicationContext(), "Please Enter Text....", Toast.LENGTH_LONG).show();
78           }
79       }
